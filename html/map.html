Caption: Map of sampling stations visited From 2014 to 2020. (A) The sampling area off the continental margin of SW Taiwan. (B) Sampling sites on the upper Gaoping Submarine Canyon (GC) and Gaoping Slope (GS),abbreviated as GC1 and GS1, respectively.

rm(list=ls())
library(marmap)
library(ggplot2)
library(dplyr)
library(viridis)
library(readxl)
library(ggrepel)
library(ggsn)
library(ggOceanMapsData)
library(ggOceanMaps)
library(patchwork)

dt <- data.frame(lon = c(120.3, 120.3, 120.5, 120.5), lat = c(22.2, 22.5, 22.5, 22.2))

tw<-basemap(limits = c(119, 123, 21, 26), bathymetry = TRUE)+ 
  scale_fill_viridis_d("Depth (m)",option = "G",direction = -1)+
  geom_polygon(data =dt, aes(x = lon, y = lat), 
               color = "red", fill = NA)+ 
  labs(x ="Longitude", y = "Latitude")+
  annotation_scale(location = "tr")+
  geom_text(x = 120.2, y =22.1, 
            colour = "red", size = 3.5, 
            aes(label = "Study area"))+
  labs(title ="A")+
  theme_bw()

long_GC1 <- 120.4105
long_GS1 <- 120.3941
lat_GC1 <- 22.41653
lat_GS1 <- 22.24033
GPSC<-read.table('119_121_21_23.xyz',
                 col.names = c("long", "lat", "depth")) %>%
  data.frame %>% 
  filter(long > 120.25 & long < 120.7) %>% 
  filter(lat<22.7& lat > 22.1)

#plot

site<- ggplot()+
  xlab("Longitude")+ylab("Latitude")+
  geom_tile(data = GPSC[GPSC$depth >= 0,], 
            aes(x = long, y = lat), fill = "grey") +
  geom_raster(data = GPSC[GPSC$depth < 0 & GPSC$depth > -1000 ,],
              aes(x =long, y = lat, fill = depth)) +
  scale_fill_viridis(option = "G",limits = c(-1000, 0),
                     breaks = seq(0,-1000, -100),
                     labels = abs) +
  geom_contour(data =GPSC, aes(x=long, y=lat, z=depth),
               breaks=0,size=0.8,colour="black")+
  geom_contour(data =GPSC, aes(x=long, y=lat, z=depth),
               breaks=-100,size=0.3,colour="lightgrey")+
  geom_contour(data =GPSC, aes(x=long, y=lat, z=depth),
               breaks=-300,size=0.3,colour="lightgrey")+
  geom_contour(data =GPSC, aes(x=long, y=lat, z=depth),
               breaks=-500,size=0.3,colour="lightgrey")+
  geom_contour(data =GPSC, aes(x=long, y=lat, z=depth),
               breaks=-700,size=0.3,colour="lightgrey")+
  geom_point(aes(x = long_GC1, y = lat_GC1),color="black")+
  geom_text_repel(aes(x = long_GC1, y = lat_GC1, label = "GC1"), 
                  fontface = "bold", size=5,
                  nudge_x = c(-0.1), nudge_y = c(0.05)) +
  geom_point(aes(x = long_GS1, y = lat_GS1),color="black")+
  geom_text_repel(aes(x = long_GS1, y = lat_GS1, label = "GS1"), 
                  fontface = "bold", size=5,
                  nudge_x = c(0.1), nudge_y = c(0.05)) +
  coord_fixed(expand = FALSE) +
  labs(fill = "Depth (m)")+
  scale_x_continuous(breaks = seq(120.3, 120.6, 0.1), 
                     labels=c("120.3E","120.4E","120.5E","120.6E"))+
  scale_y_continuous(breaks = seq(22.2, 22.6, 0.1),
                     labels=c("22.2N","22.3N","22.4N","22.5N","22.6N"))+
  scalebar(x.min=120.25, x.max=120.7, y.min=22.1, y.max= 22.7,
           model = "WGS84",  dist_unit = "km", transform = TRUE,
           location = "topright", anchor = c(x = 120.66, y = 22.68),
           height = 0.015, dist = 5, st.size = 3, border.size = 0.1) +
  labs(title ="B")+
  theme_bw()
map<-tw+site 
ggsave("map.png",map,width = 12, height =9)
